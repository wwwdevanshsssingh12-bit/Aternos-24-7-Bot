local function getTargets(str)
	local t={} str=str and str:lower() or "me"
	if str=="all" then return PS:GetPlayers() end
	if str=="others" then for _,p in ipairs(PS:GetPlayers()) do if p~=LP then table.insert(t,p) end end return t end
	if str=="me" or str=="" then return {LP} end
	if str=="random" then local plrs=PS:GetPlayers() return {plrs[math.random(1,#plrs)]} end
	if str=="friends" then for _,p in ipairs(PS:GetPlayers()) do if p:IsFriendsWith(LP.UserId) then table.insert(t,p) end end return t end
	if str=="nonfriends" then for _,p in ipairs(PS:GetPlayers()) do if not p:IsFriendsWith(LP.UserId) then table.insert(t,p) end end return t end
	if str=="alive" then for _,p in ipairs(PS:GetPlayers()) do if p.Character and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health>0 then table.insert(t,p) end end return t end
	if str=="dead" then for _,p in ipairs(PS:GetPlayers()) do if p.Character and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health<=0 then table.insert(t,p) end end return t end
	if str=="nearest" then local n,d=nil,math.huge for _,p in ipairs(PS:GetPlayers()) do if p~=LP and getHRP(p) and getHRP() then local mag=(getHRP().Position-getHRP(p).Position).Magnitude if mag<d then n=p d=mag end end end if n then table.insert(t,n) end return t end
	if str=="farthest" then local n,d=nil,0 for _,p in ipairs(PS:GetPlayers()) do if p~=LP and getHRP(p) and getHRP() then local mag=(getHRP().Position-getHRP(p).Position).Magnitude if mag>d then n=p d=mag end end end if n then table.insert(t,n) end return t end
	if str:sub(1,1)=="%" then local tm=str:sub(2) for _,p in ipairs(PS:GetPlayers()) do if p.Team and p.Team.Name:lower():find(tm) then table.insert(t,p) end end return t end
	for _,p in ipairs(PS:GetPlayers()) do if p.Name:lower():sub(1,#str)==str or p.DisplayName:lower():sub(1,#str)==str then table.insert(t,p) end end return t
end
local function T(a, f) for _, p in ipairs(getTargets(a)) do pcall(f, p, getChar(p), getHRP(p), getHum(p)) end end
